using System;
using System.Linq;
using System.Collections.Generic;
using System.IO;

namespace cbradioOOP {
    class Program {
        public static List<Adasok> adasok = new List<Adasok>();
        static void Main(string[] args) {
            StreamReader olvaso = new StreamReader("cb.txt");
            string fejlec = olvaso.ReadLine();
            while(!olvaso.EndOfStream) {
                adasok.Add(new Adasok(olvaso.ReadLine()));
            }
            Console.WriteLine("3. feladat: Bejegyzések száma: {0} db",adasok.Count);
            for(int i=0; i<adasok.Count; i++) {
                if(adasok[i].AdasDb==4) {
                    Console.WriteLine("4. feladat: Volt négy adást indító sofőr.");
                    break;
                }
            }
            Console.ReadKey();
        }
    }

    class Adasok {
        public int Ora { get; set; }
        public int Perc { get; set; }
        public int AdasDb { get; set; }
        public string Nev { get; set; }
        public Adasok(string sor) {
            string [] sor2 = sor.Split(';');
            Ora = Convert.ToInt32(sor2[0]);
            Perc = Convert.ToInt32(sor2[1]);
            AdasDb = Convert.ToInt32(sor2[2]);
            Nev = sor2[3];
        }
    }
}
